#include <iostream>
#include <queue>
#include <functional>
using namespace std;

int main(){
	ios::sync_with_stdio(false);
	cin.tie(0);

	int n, ans, temp, a, b;
    priority_queue<int, vector<int>, greater<int>> num;
	while (cin >> n && n){
		ans = 0;
		for (int i = 0; i < n; ++i){
			cin >> temp;
			num.push(temp);
		}
		for (int i = 0; i < n - 1; ++i){
            a = num.top(); num.pop();
            b = num.top(); num.pop();
			num.push(a+b);
            ans += a + b;
		}
		cout << ans << '\n';
        num.pop();
	}
	return 0;
}
